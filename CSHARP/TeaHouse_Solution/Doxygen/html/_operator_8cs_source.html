210  private void SupportCreateOption(string tableName)
211  {
212  string s = tableName + ";";
213  switch (tableName)
214  {
215  case "TEA":
216  GetConsoleLogic().ReadTea();
217  Console.WriteLine("Add new tea's data separated by semicolons like the following example:\n" +
218  "TeaID;TeaName;Quantity;BasePrice;Sales;Total\n");
219  s += Console.ReadLine();
220  GetConsoleLogic().CreateTea(s);
221  break;
222 
223  case "TEALEAF":
224  Console.WriteLine("Add new tealeaf's data separated by semicolons like the following example:\n" +
225  "TealeafID;TealeafName;Variety;Brand;Acidity;Flavour;Region\n");
226  s += Console.ReadLine();
227  GetConsoleLogic().CreateTealeaf(s);
228  break;
229 
230  case "EXTRA":
231  Console.WriteLine("Add new extra's data separated by semicolons like the following example:\n" +
232  "ExtraID;ExtraName;Category;Allergen;Availability;Price\n");
233  s += Console.ReadLine();
234  GetConsoleLogic().CreateExtra(s);
235  break;
236  case "SUPPLIER":
237  Console.WriteLine("Add new supplier's data separated by semicolons like the following example:\n" +
238  "SupplierID;SupplierName;Street;City;OrderStatus;TotalOrders\n");
239  s += Console.ReadLine();
240  GetConsoleLogic().CreateSupplier(s);
241  break;
242  default: Console.WriteLine("Invalid table name!\n Tables: TEA, TEALEAF, EXTRA, SUPPLIER");
243  break;
244  }
